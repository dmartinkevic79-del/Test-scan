async function initScanner() { if (html5QrCode) await html5QrCode.stop().catch(() => {}); html5QrCode = new Html5Qrcode("reader"); // Упрощенный конфиг для максимальной совместимости const config = { fps: 25, qrbox: { width: 280, height: 140 }, aspectRatio: 1.0 }; try { await html5QrCode.start( { facingMode: "environment" }, config, (text) => { const now = Date.now(); if (now - lastScanTime < SCAN_DELAY) return; let match = text.match(/\d{10,}/); let processed = match ? match[0] : text; if (!codes.some(c => c.text === processed)) { const timestamp = new Date(); codes.push({ text: processed, time: timestamp.toLocaleTimeString(), date: timestamp.toLocaleDateString() }); localStorage.setItem('wb_codes_v5', JSON.stringify(codes)); triggerVisualFeedback(); renderList(); new Audio('https://assets.mixkit.co/active_storage/sfx/2568/2568-preview.mp3').play().catch(()=>{}); } lastScanTime = now; } ); document.getElementById('start-btn').style.display = 'none'; setTimeout(checkTorch, 1000); } catch (err) { // Выводим детальную ошибку для диагностики console.error(err); if (err.includes("NotAllowedError")) { alert("Ошибка: Вы запретили доступ к камере в настройках браузера."); } else { alert("Ошибка камеры: " + err + "\n\nПопробуйте открыть сайт в обычном браузере (Chrome/Safari), а не через мессенджер."); } } }
